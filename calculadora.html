// Função que cria um bipe eletrônico "retro" function tocarSom() { const contexto = new (window.AudioContext || window.webkitAudioContext)(); const oscilador = contexto.createOscillator(); const ganho = contexto.createGain(); oscilador.type = 'sine'; // Tipo de onda (suave) oscilador.frequency.setValueAtTime(440, contexto.currentTime); // Nota Lá (440Hz) ganho.gain.setValueAtTime(0.1, contexto.currentTime); // Volume baixo ganho.gain.exponentialRampToValueAtTime(0.01, contexto.currentTime + 0.1); oscilador.connect(ganho); ganho.connect(contexto.destination); oscilador.start(); oscilador.stop(contexto.currentTime + 0.1); // Dura apenas 0.1 segundos }
